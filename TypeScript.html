TypeScript Cheat Sheet

Basic Types

// Boolean
let isDone: boolean = false;

// Number
let age: number = 25;
let height: number = 175.5;

// String
let name: string = "John Doe";

// Array
let numbers: number[] = [1, 2, 3];
let names: Array<string> = ["Alice", "Bob", "Charlie"];

// Tuple
let person: [string, number] = ["John Doe", 25];

// Enum
enum Color {
  Red,
  Green,
  Blue,
}
let myColor: Color = Color.Green;

// Any
let notSure: any = 4;
notSure = "maybe a string";

// Void
function logMessage(): void {
  console.log("Hello, World!");
}

// Null and Undefined
let someValue: null = null;
let otherValue: undefined = undefined;

// Object
let obj: object = { key: "value" };

// Type Assertions
let someValue: any = "this is a string";
let strLength: number = (<string>someValue).length;
let strLength: number = (someValue as string).length;

Functions

// Function Declaration
function add(a: number, b: number): number {
  return a + b;
}

// Function Expression
const multiply = function (a: number, b: number): number {
  return a * b;
};

// Arrow Function
const subtract = (a: number, b: number): number => a - b;

// Optional Parameters
function greet(name: string, age?: number): void {
  console.log(`Hello, ${name}! You are ${age} years old.`);
}
greet("John");
greet("Alice", 25);

// Default Parameters
function multiply(a: number, b: number = 2): number {
  return a * b;
}
multiply(5); // Returns 10

// Rest Parameters
function sum(...numbers: number[]): number {
  return numbers.reduce((total, num) => total + num, 0);
}
sum(1, 2, 3, 4); // Returns 10

// Function Overloads
function processInput(value: string): string;
function processInput(value: number): number;
function processInput(value: string | number): string | number {
  if (typeof value === "string") {
    return value.toUpperCase();
  }
  return value * 2;
}

Interfaces

interface Person {
  name: string;
  age: number;
}

const person: Person = {
  name: "John Doe",
  age: 25,
};

interface Shape {
  readonly id: number;
  color?: string;
  getArea(): number;
}

class Circle implements Shape {
  readonly id: number;
  radius: number;

  constructor(id: number, radius: number) {
    this.id = id;
    this.radius = radius;
  }

  getArea(): number {
    return Math.PI * this.radius ** 2;
  }
}

Classes

class Animal {
  protected name: string;

  constructor(name: string) {
    this.name = name;
  }

  move(distance: number = 0): void {
    console.log(`${this.name} moved ${distance} meters.`);
  }
}

class Dog extends Animal {
  bark(): void {
    console.log("Woof! Woof!");
  }
}

const dog = new Dog("Fido");
dog.move(10); // Output: "Fido moved 10 meters."
dog.bark(); // Output: "Woof! Woof!"

Generics

function identity<T>(arg: T): T {
  return arg;
}

let result = identity<string>("Hello, World!");
let result = identity("Hello, World!"); // Type inference

interface Box<T> {
  value: T;
}

let box: Box<number> = { value: 42 };

Modules

// Exporting
export const PI = Math.PI;

export function double(num: number): number {
  return num * 2;
}

export default class MyClass {
  // ...
}

// Importing
import { PI, double } from "./math";
import MyClass from "./MyClass";
